## 多机分布式通信
```bash
################################ROS从机配置#########################
# ~/.bashrc
export ROS_HOSTNAME=172.16.10.244   #从机ip
export ROS_MASTER_URI=http://172.16.10.181:11311  #主机ip
export ROS_IP=172.16.10.181  #主机ip

# /etc/hosts
172.16.10.181 giraffe  #主机ip+用户名
172.16.10.244 photinia  #从机ip+用户名
################################END################################

################################ROS主机配置#########################
#主机配置 ~/.bashrc
export ROS_HOSTNAME=172.16.10.181   #主机ip
export ROS_MASTER_URI=http://172.16.10.181:11311  #主机ip
export ROS_IP=172.16.10.181  #主机ip

# /etc/hosts
172.16.10.181 giraffe  #主机ip+用户名
172.16.10.244 photinia  #从机ip+用户名
################################END################################
```
